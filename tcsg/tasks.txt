import (
	"fmt"

	"os"
	"strings"

func reverse(s string) string {
	if len(s) <= 1 {
		return s
	}

	arr := make([]byte, len(s))

	for i, j := 0, len(s)-1; i < j; {
		arr[i], arr[j] = s[j], s[i]
		i++
		j--
	}
	return string(arr)
}

func main() {
	b, _ := io.ReadAll(os.Stdin)
	str := strings.Trim(string(b), "\n")

	result := reverse(str)

	fmt.Fprintln(os.Stdout, result)
}
